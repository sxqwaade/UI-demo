javascript

$j(function(){
  var comparison = $j(".product-comparison");
  var k = 4;
  var i = 0;
  var q = 0;
  comparison.click(function(){
    var c_pic = $j(this).siblings(".col-pic").find("img").attr("data-comparison");
    var c_name = $j(this).siblings(".col-txt").find(".col-name a").text();
    var c_box = "<div class='c_pic'><img src='"+c_pic+"' width='190px' height='190px' alt='"+c_name+"' title='"+c_name+"' /></div><h6 class='c_name'>"+c_name+"</h6><a class='fix-png c_del' href='javascript:;' title='删除'>删除</a>";
    if($j(this).attr("c_id")){
      var cid = $j(this).attr("c_id");
      $j(this).removeAttr("c_id").find("a").removeClass("checked").text("与其他型号进行对比");
      $j(".comparison-list").find("li[c_id='"+cid+"']").removeClass("lcom").removeAttr("c_id").fadeOut(400,function(){
        $j(this).empty().appendTo($j(this).parent()).show();
      });
      i--;
      // 2个以上可以点击开始对比按钮
      if($j(".comparison-list li[c_id]").length>=2){
        $j(".comparison-btn").removeClass("uncom");
      }else{
        $j(".comparison-btn").addClass("uncom")
        return false;
      }
    }else{
      if(i<k){
        $j(this).attr("c_id",q).find("a").addClass("checked").text("移除对比");
        $j(".comparison-list li").eq(i).addClass("lcom").attr("c_id",q).append($j(c_box));
        i++;
        q++;
        // 2个以上可以点击开始对比按钮
        if($j(".comparison-list li[c_id]").length>=2){
          $j(".comparison-btn").removeClass("uncom");
        }else{
          $j(".comparison-btn").addClass("uncom")
          return false;
        }
      }
    }
  });
  var c_del = $j(".c_del");
  c_del.live(
    'click',function(){
      var cid = $j(this).parent().attr("c_id");
      $j(this).parent().removeAttr("c_id").fadeOut(400,function(){
        $j(this).removeClass("lcom").empty().appendTo($j(this).parent()).show();
        $j(".cate-list-fall").find(".product-comparison[c_id="+cid+"]").removeAttr("c_id").find("a").removeClass("checked").text("与其他型号进行对比");
      });
      i--;
      // 2个以上可以点击开始对比按钮
      if($j(".comparison-list li[c_id]").length>=2){
        $j(".comparison-btn").removeClass("uncom");
      }else{
        $j(".comparison-btn").addClass("uncom")
        return false;
      }
    }
  )
})
