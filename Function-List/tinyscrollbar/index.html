javascript

tinyscrollbar 支持手持设备。 具体看http://baijs.nl/tinyscrollbar/

兼容IE6

var $j = jQuery.noConflict();
$j(function(){
  var cli = $j(".cart-list-li");
  if(cli.length>0){
    if(jQuery.browser.msie&&jQuery.browser.version==6){
      $j(".scrollbar").remove();
      $j(".cart-list-ul").css({"position":"static","width":600});
      $j(".viewport").css("overflowY","scroll");
    }else{
      $j(".viewport").css("overflow","hidden");
      // 自定义滚动条
      $j('.cs-scroll').tinyscrollbar();
    }
   }
})
